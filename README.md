 <h1>Without Auto Slide</h1>
         <pre>
			$(".cust_slider").loadCustomSlider(
					{
						autoslide:false
					});
		</pre>
		<h1>With Auto Slide</h1>
         <pre>
		$(".cust_slider2").loadCustomSlider(
				{
					speed:3000,

				});
		</pre>
